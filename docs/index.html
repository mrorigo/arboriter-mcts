Simple, clean API

Integrating MCTS into your game is straightforward. Configure your search parameters, create the engine, and find the best move.

Read the Docs ‚Üí 
use arboriter_mcts::{MCTS, MCTSConfig};

// 1. Configure
let config = MCTSConfig::default()
    .with_exploration_constant(1.414)
    .with_max_iterations(10_000);

// 2. Initialize
let mut mcts = MCTS::new(initial_state, config);

// 3. Search
let best_action = mcts.search()?;

println!("Best move: {:?}", best_action);
